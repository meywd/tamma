```vue
<template>
  <div class="interactive-example">
    <div class="example-header">
      <h4>{{ title }}</h4>
      <div class="controls">
        <button @click="runCode" :disabled="running" class="run-button">
          {{ running ? 'Running...' : 'Run Code' }}
        </button>
        <button @click="resetCode" class="reset-button">Reset</button>
        <select v-model="selectedLanguage" class="language-select">
          <option v-for="lang in languages" :key="lang" :value="lang">
            {{ lang }}
          </option>
        </select>
      </div>
    </div>

    <div class="editor-container">
      <div class="code-editor">
        <textarea
          v-model="code"
          :placeholder="placeholder"
          class="code-textarea"
          spellcheck="false"
        ></textarea>
      </div>

      <div class="output-container">
        <div class="output-header">
          <span>Output</span>
          <button @click="clearOutput" class="clear-button">Clear</button>
        </div>
        <div class="output-content">
          <pre v-if="output">{{ output }}</pre>
          <div v-else class="output-placeholder">Run code to see output...</div>
        </div>
      </div>
    </div>

    <div v-if="error" class="error-message"><strong>Error:</strong> {{ error }}</div>
  </div>
</template>

<script setup lang="ts">
import { ref, computed } from 'vue';

interface Props {
  title: string;
  initialCode: Record<string, string>;
  placeholder?: string;
  languages?: string[];
}

const props = withDefaults(defineProps<Props>(), {
  placeholder: 'Enter your code here...',
  languages: () => ['typescript', 'javascript', 'python'],
});

const code = ref(props.initialCode[props.languages[0]] || '');
const output = ref('');
const error = ref('');
const running = ref(false);
const selectedLanguage = ref(props.languages[0]);

const runCode = async () => {
  running.value = true;
  error.value = '';
  output.value = '';

  try {
    const response = await fetch('/api/execute-code', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        language: selectedLanguage.value,
        code: code.value,
      }),
    });

    const result = await response.json();

    if (result.success) {
      output.value = result.output;
    } else {
      error.value = result.error;
    }
  } catch (err) {
    error.value = err instanceof Error ? err.message : 'Unknown error';
  } finally {
    running.value = false;
  }
};

const resetCode = () => {
  code.value = props.initialCode[selectedLanguage.value] || '';
  clearOutput();
};

const clearOutput = () => {
  output.value = '';
  error.value = '';
};
</script>
```

```typescript
// docs/.vitepress/search/index.ts
import type { MiniSearchOptions } from 'minisearch';

export interface SearchResult {
  id: string;
  title: string;
  text: string;
  url: string;
  excerpt: string;
}

export class DocumentationSearch {
  private miniSearch: any;
  private documents: any[] = [];

  constructor() {
    // Dynamic import to avoid SSR issues
    import('minisearch').then(({ MiniSearch }) => {
      this.miniSearch = new MiniSearch({
        fields: ['title', 'text', 'headings'],
        storeFields: ['title', 'text', 'url'],
        searchOptions: {
          boost: { title: 2, headings: 1.5 },
          fuzzy: 0.2,
          prefix: true,
        },
      } as MiniSearchOptions);
    });
  }

  async indexDocuments() {
    // Load all markdown files
    const docs = await this.loadAllDocuments();

    this.documents = docs.map((doc) => ({
      id: doc.url,
      title: doc.title,
      text: doc.content,
      headings: doc.headings.join(' '),
      url: doc.url,
    }));

    this.miniSearch.addAll(this.documents);
  }

  search(query: string): SearchResult[] {
    if (!this.miniSearch) return [];

    const results = this.miniSearch.search(query);

    return results
      .map((result: any) => {
        const doc = this.documents.find((d) => d.id === result.id);
        if (!doc) return null;

        return {
          id: result.id,
          title: doc.title,
          text: doc.text,
          url: doc.url,
          excerpt: this.generateExcerpt(doc.text, query),
          score: result.score,
        };
      })
      .filter(Boolean);
  }

  private generateExcerpt(text: string, query: string): string {
    const words = query.toLowerCase().split(' ');
    const index = text.toLowerCase().indexOf(words[0]);

    if (index === -1) return text.substring(0, 150) + '...';

    const start = Math.max(0, index - 50);
    const end = Math.min(text.length, index + 150);

    let excerpt = text.substring(start, end);

    if (start > 0) excerpt = '...' + excerpt;
    if (end < text.length) excerpt = excerpt + '...';

    // Highlight query terms
    words.forEach((word) => {
      const regex = new RegExp(`(${word})`, 'gi');
      excerpt = excerpt.replace(regex, '<mark>$1</mark>');
    });

    return excerpt;
  }

  private async loadAllDocuments(): Promise<any[]> {

  }
}
```

### 5. Analytics and Feedback

```typescript
// docs/.vitepress/analytics.ts
export interface AnalyticsEvent {
  type: 'page_view' | 'search' | 'example_run' | 'feedback';
  data: Record<string, any>;
  timestamp: number;
  sessionId: string;
}

export class DocumentationAnalytics {
  private sessionId: string;
  private events: AnalyticsEvent[] = [];

  constructor() {
    this.sessionId = this.generateSessionId();
    this.initEventListeners();
  }

  private generateSessionId(): string {
    return Date.now().toString(36) + Math.random().toString(36).substr(2);
  }

  private initEventListeners() {
    // Page view tracking
    if (typeof window !== 'undefined') {
      this.trackPageView();

      // Track page changes
      window.addEventListener('popstate', () => {
        this.trackPageView();
      });
    }
  }

  trackPageView() {
    this.sendEvent({
      type: 'page_view',
      data: {
        url: window.location.pathname,
        title: document.title,
        referrer: document.referrer,
      },
    });
  }

  trackSearch(query: string, results: number) {
    this.sendEvent({
      type: 'search',
      data: {
        query,
        resultsCount: results,
      },
    });
  }

  trackExampleRun(exampleId: string, language: string, success: boolean) {
    this.sendEvent({
      type: 'example_run',
      data: {
        exampleId,
        language,
        success,
      },
    });
  }

  trackFeedback(page: string, rating: number, comment?: string) {
    this.sendEvent({
      type: 'feedback',
      data: {
        page,
        rating,
        comment,
      },
    });
  }

  private sendEvent(event: Omit<AnalyticsEvent, 'timestamp' | 'sessionId'>) {
    const fullEvent: AnalyticsEvent = {
      ...event,
      timestamp: Date.now(),
      sessionId: this.sessionId,
    };

    this.events.push(fullEvent);

    // Send to analytics service
    this.flushEvents();
  }

  private async flushEvents() {
    if (this.events.length === 0) return;

    const eventsToSend = [...this.events];
    this.events = [];

    try {
      await fetch('/api/analytics', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({ events: eventsToSend }),
      });
    } catch (error) {
      // Re-add events on failure
      this.events.unshift(...eventsToSend);
    }
  }
}
```

````typescript
        if (url && url.startsWith('/') && !url.startsWith('http')) {
          const expectedPath = url.slice(1).replace(/\.html$/, '.md');
````

```typescript
      expect(metrics.lcp).toBeLessThan(2500); // Largest Contentful Paint
      expect(metrics.fid).toBeLessThan(100); // First Input Delay
      expect(metrics.cls).toBeLessThan(0.1); // Cumulative Layout Shift
    }
  });

  test('Search performance', async () => {
    const searchIndex = await loadSearchIndex();

    const start = performance.now();
    const results = searchIndex.search('autonomous development');
    const duration = performance.now() - start;

    expect(duration).toBeLessThan(100); // Search should be fast
    expect(results.length).toBeGreaterThan(0);
  });

  test('Bundle size optimization', async () => {
    const stats = await getBuildStats();

    expect(stats.totalSize).toBeLessThan(1024 * 1024); // < 1MB total
    expect(stats.vendorSize).toBeLessThan(300 * 1024); // < 300KB vendor
  });
});
```
